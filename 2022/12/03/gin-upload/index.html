【GIN学习笔记】Gin框架文件的上传

发布于 2022-12-03 GIN学习笔记  golang gin 

  1. 首先简单写一个HTML页面实现文件上传的交互：

    <!DOCTYPE html>
<html lang="en">
    <head>
        <title>upload demo</title>
    </head>
    <body>
        <form action="/upload", method="post", enctype="multipart/form-data">
            <input type="file" name="file">
            <input type="submit" value="upload">
        </form>
    </body>
</html>

  2. 然后在写一个GET方法，当访问路径/upload时显示该网页：

    router.LoadHTMLFiles("./index.html")
router.GET("/upload", func(ctx *gin.Context) {
    ctx.HTML(http.StatusOK, "index.html", nil)
})

  3. 接着处理前端传来的POST请求：

    • 单文件上传
    router.POST("/upload", func(ctx *gin.Context) {
    // 获取文件数据
    file, err := ctx.FormFile("file")
    // 异常处理
    if err != nil {
        ctx.JSON(http.StatusBadRequest, gin.H{
            "error": err.Error(),
        })
        return
    }
    log.Println(file.Filename)
    dst := path.Join("./", file.Filename)
    // 保存文件
    ctx.SaveUploadedFile(file, dst)
    ctx.JSON(http.StatusOK, gin.H{
        "message": fmt.Sprintf("%s upload success!", file.Filename),
    })
})

    • 多文件上传

      为了实现多文件上传需要对HTML页面的file类型的input标签添加multiple属性：

    <form action="/upload", method="post", enctype="multipart/form-data">
    <!-- 添加multiple属性-->
    <input type="file" name="files" multiple>
    <input type="submit" value="upload">
</form>

    ​ 和单文件上传不同，多文件上传需要使用MultipartForm()方法获取form数据，然后遍历所有文件数据并且依次保存，保存方法和单文件示例类似：

    router.POST("/multi-upload", func(ctx *gin.Context) {
    // 使用MultipartForm()处理提交的文件时，默认的内存限制为32MB
    form, err := ctx.MultipartForm()
    // 异常处理
    if err != nil {
        if err != nil {
            ctx.JSON(http.StatusBadRequest, gin.H{
                "error": err.Error(),
            })
            return
        }
    }
    // key要对应input标签的name值
    files := form.File["files"]
    
    for _, file := range files {
        log.Println(file.Filename)
        dst := path.Join("./", file.Filename)
        ctx.SaveUploadedFile(file, dst)
    }
    ctx.JSON(http.StatusOK, gin.H{
        "message": fmt.Sprintf("%d upload success!", len(files)),
    })
})
